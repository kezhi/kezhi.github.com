gulp-coffee-demo

Step

  1. npm init 创建package.json文件

  2. 安装所需模块

    npm install --save gulp

    npm install --save gulp-coffee

    npm install --save gulp-coffeelint

  3. 创建 src build 文件夹，并在src里创建一个.coffee文件

  4. 创建 Gulpfile.js

     var gulp = require('gulp');
 var coffee = require('gulp-coffee');
 var coffeelint = require('gulp-coffeelint');
 var coffeescript_files = ['./src/**/*.coffee'];
 var build_dir = './build/';

 //校验coffee语法的任务
  gulp.task('validate_coffee', function () {
       gulp.src(coffeescript_files)
     .pipe(coffeelint())
        .pipe(coffeelint.reporter());
 });

 //把coffee脚本编译为js的任务
 gulp.task('compile_coffee', ['validate_coffee'], function() {
       gulp.src(coffeescript_files)
     .pipe(gulp.dest(build_dir));
 });
 gulp.task('default', ['compile_coffee']);

  5. 在命令行执行 gulp
