Express 文件上传 - Multer

Jun 10 2015

  1. 安装所需模块

    npm install --save multer

  2. 用法，在app.js添加以下代码：

    var express = require('express')
var multer  = require('multer')
var app = express()

app.use(multer({
      dest: './uploads/',
       rename: function (fieldname, filename) {
            return filename.replace(/\W+/g, '-').toLowerCase() + Date.now()
      }
}))

  3. 路由模块

    router.post('/uploadimage',function(req,res){
    res.status(200).json({
        file:req.files
    });
});

  4. 上传部分的html代码

    <form id='editfile' method='post' action='/uploadimage' enctype='multipart/form-data'>
    <input type="text"/>
    选择图片：<input id='upfile' type='file'/>
    <input type='submit' value='提交'/>
</form>

    enctype=”multipart/form-data”的作用是设置表单上传文件的正确编码，浏览器用表单提交数据的格式

    使用ajax上传：

$('#editfile').submit(function (e) {
    var data = new FormData();
    data.append('file', $('#upfile')[0].files[0]);
    $.ajax({
        url: '/uploadimage',
        data: data,
        processData: false,
        contentType: false,
        type: 'POST',
        success: function(data){
            console.log(data);   //返回上传后所在的路径
        }
    });
    e.preventDefault();
});
